def convert_base(val: str, base: int):
    if (base < 0 or (base > 10 and base != 16)):
        return -1
    value = 0
    for (i, digit) in enumerate(val):
        digit_val = convert_digit_to_int(digit)
        if not ((digit_val < 0 or digit_val > base)):
            return -1

def convert_digit_to_int(char: str):
